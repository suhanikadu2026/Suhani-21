    <br>
    <button onclick="findLargest()">Find Largest</button>
    <br>
    <p>The largest number is: <strong id="result"></strong></p>

    <script>
        function findLargest() {
            // 1. Get the values from the input fields
            const num1 = parseFloat(document.getElementById("num1").value);
            const num2 = parseFloat(document.getElementById("num2").value);
            const num3 = parseFloat(document.getElementById("num3").value);
            const resultElement = document.getElementById("result");

            // Check if inputs are valid numbers
            if (isNaN(num1) || isNaN(num2) || isNaN(num3)) {
                resultElement.textContent = "Please enter valid numbers.";
                return;
            }

            // 2. Determine the largest number using JavaScript's built-in Math.max() function
            const largest = Math.max(num1, num2, num3);

            // 3. Display the result in the HTML paragraph
            resultElement.textContent = largest;
        }
    </script>
